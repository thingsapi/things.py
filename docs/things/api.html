things.api

Module implementing the Things API.

The terms of the API aim to match those used in the Things app and Things URL Scheme. In some specific cases we instead choose to use terms that occur in the Things SQL database to closer reflect its underlying data structures. Whenever that happens, we define the new term here.

 22def tasks(uuid=None, include_items=False, **kwargs):  # noqa: C901
 23    """
 24    Read tasks into dicts.
 25
 26    Note: "task" is a technical term used in the database to refer to a
 27    to-do, project, or heading. For details, check the "type"-parameter.
 28
 29    Per default, only tasks marked as incomplete are included. If you
 30    want to include completed or canceled tasks in the result, check the
 31    "status"-parameter.
 32
 33    Parameters
 34    ----------
 35    uuid : str or None, optional
 36        Any valid task uuid. If None, then return all tasks matched.
 37
 38    include_items : bool, default False
 39        Include items contained within a task. These might include
 40        checklist items, headings, and to-dos.
 41
 42    type : {'to-do', 'heading', 'project', None}, optional
 43        Only return a specific type of task:
 44        - `'to-do'`:    may have a checklist; may be in an area and have tags.
 45        - `'project'`:  may have to-dos and headings; may be in an area and
 46                        have tags.
 47        - `'heading'`:  part of a project; groups tasks.
 48        - `None` (default): return all types of tasks.
 49
 50    status : {'incomplete', 'completed', 'canceled', None}, optional, \
 51        default 'incomplete'
 52
 53        Only include tasks matching that status. If `status == None`,
 54        then include tasks with any status value.
 55
 56    start : {'Inbox', 'Anytime', 'Someday', None}, optional
 57        Only include tasks matching that start value. If the argument is
 58        `None` (default), then include tasks with any start value.
 59
 60    area : str or bool or None, optional
 61        Any valid uuid of an area. Only include tasks matching that area.
 62        Special cases:
 63        - `area == False`, only include tasks _without_ an area.
 64        - `area == True`, only include tasks _with_ an area.
 65        - `area == None` (default), then include all tasks.
 66
 67    project : str or bool or None, optional
 68        Any valid uuid of a project. Only include tasks matching that project.
 69        Special cases:
 70        - `project == False`, only include tasks _without_ a project.
 71        - `project == True`, only include tasks _with_ a project.
 72        - `project == None` (default), then include all tasks.
 73
 74    heading : str or None, optional
 75        Any valid uuid of a heading. Only include tasks matching that heading.
 76        Special cases:
 77        - `heading == False`, only include tasks _without_ a heading.
 78        - `heading == True`, only include tasks _with_ a heading.
 79        - `heading == None` (default), then include all tasks.
 80
 81    tag : str or bool or None, optional
 82        Any valid title of a tag. Only include tasks matching that tag.
 83        Special cases:
 84        - `tag == False`, only include tasks _without_ tags.
 85        - `tag == True`, only include tasks _with_ tags.
 86        - `tag == None` (default), then include all tasks.
 87
 88    start_date : bool, str or None, optional
 89        - `start_date == False`, only include tasks _without_ a start date.
 90        - `start_date == True`, only include tasks with a start date.
 91        - `start_date == 'future'`, only include tasks with a future start date.
 92        - `start_date == 'past'`, only include tasks with a past start date.
 93          Note: this includes today's date.
 94        - `start_date == None` (default), then include all tasks.
 95
 96    stop_date : as start_date, signifies the date of completion
 97
 98    exact : bool, optional
 99        - `exact == False` (default) compare date provided with >=
100        - `exact == True`, compare date with ==
101
102    deadline : bool, str or None, optional
103        - `deadline == False`, only include tasks _without_ a deadline.
104        - `deadline == True`, only include tasks _with_ a deadline.
105        - `deadline == 'future'`, only include tasks with a future deadline.
106        - `deadline == 'past'`, only include tasks with a past deadline.
107          Note: this includes today's date.
108        - `deadline == None` (default), then include all tasks.
109
110    deadline_suppressed : bool or None, optional
111        "deadline suppressed" is a technical term used in the database.
112        When tasks have an overdue deadline they show up in Today.
113        Some of us "suppress" some tasks, that is, move them out of
114        Today and into Inbox, Anytime, or Someday. For those moved tasks
115        the deadline is said to be suppressed.
116        - `deadline_suppressed == True`, only include tasks with deadlines
117          and whose deadlines have been suppressed.
118        - `deadline_suppressed == False`, include any task _except_ those
119          with suppressed deadlines.
120        - `deadline_suppressed == None` (default), include any task.
121
122    trashed : bool or None, optional, default False
123        - `trashed == False` (default), only include non-trashed tasks.
124        - `trashed == True`, only include trashed tasks.
125        - `trashed == None`, include both kind of tasks.
126
127    context_trashed : bool or None, optional, default False
128        Some tasks may be within a context of a project or a heading.
129        This manages how to handle such tasks. The other tasks are not
130        affected by this setting.
131
132        - `context_trashed == False` (default), for tasks within a context,
133           only return tasks whose context has _not_ been trashed.
134        - `context_trashed == True`, for tasks within a context,
135           only return tasks whose context has been trashed.
136        - `context_trashed == None`, include both kind of tasks.
137
138    last : str, optional
139        Limit returned tasks to tasks created within the last X days,
140        weeks, or years. For example: '3d', '5w', or '1y'.
141
142        Takes as input an offset string of the format 'X[d/w/y]' where X
143        is a non-negative  integer that is followed by 'd', 'w', or 'y'
144        which stand for 'days', 'weeks', and 'years', respectively.
145
146    search_query : str, optional
147        The string value is passed to the SQL LIKE operator. It can thus
148        include placeholders such as '%' and '_'. Per default, it is
149        wrapped in '% ... %'.
150
151        Currently titles and notes of to-dos, projects, headings, and areas
152        are taken into account.
153
154    index : {'index', 'todayIndex'}, default 'index'
155        Database field to order result by.
156
157    count_only : bool, default False
158        Only output length of result. This is done by a SQL COUNT query.
159
160    print_sql : bool, default False
161        Print every SQL query performed. Some may contain '?' and ':'
162        characters, which correspond to SQLite parameter tokens.
163        See https://www.sqlite.org/lang_expr.html#varparam
164
165    filepath : str, optional
166        Any valid path of a SQLite database file generated by the Things app.
167        If the environment variable `THINGSDB` is set, then use that path.
168        Otherwise, access the default database path.
169
170    database : things.database.Database, optional
171        Any valid database object previously instantiated.
172
173    Returns
174    -------
175    list of dict (default)
176        Representing multiple tasks.
177    dict (if `uuid` is given)
178        Representing a single task.
179    int (`count_only == True`)
180        Count of matching Tasks.
181
182    Examples
183    --------
184    >>> things.tasks()
185    [{'uuid': '6Hf2qWBjWhq7B1xszwdo34', 'type': 'to-do', 'title':...
186    >>> things.tasks('DfYoiXcNLQssk9DkSoJV3Y')
187    {'uuid': 'DfYoiXcNLQssk9DkSoJV3Y', 'type': 'to-do', 'title': ...
188    >>> things.tasks(area='hIo1FJlAYGKt1Yj38vzKc3', include_items=True)
189    []
190    >>> things.tasks(status='completed', count_only=True)
191    10
192    >>> things.tasks(status='completed', last='1w', count_only=True)
193    0
194
195    """
196    database = pop_database(kwargs)
197    result = database.get_tasks(
198        uuid=uuid, status=kwargs.pop("status", "incomplete"), **kwargs
199    )
200
201    if kwargs.get("count_only"):
202        return result
203
204    # overwrite `include_items` if fetching a single task.
205    if uuid:
206        include_items = True
207
208    for task in result:
209        # TK: How costly of an operation is it to do this for every task?
210        # IF costly, then can it be made significantly more efficient
211        # by optimizing SQL calls?
212
213        if task.get("tags"):
214            task["tags"] = database.get_tags(task=task["uuid"])
215
216        if not include_items:
217            continue
218
219        # include items
220        if task["type"] == "to-do":
221            if task.get("checklist"):
222                task["checklist"] = database.get_checklist_items(task["uuid"])
223        elif task["type"] == "project":
224            project = task
225            project["items"] = items = tasks(
226                project=project["uuid"],
227                context_trashed=None,
228                include_items=True,
229                database=database,
230            )
231            # to-dos without headings appear before headings in app
232            items.sort(key=lambda item: item["type"], reverse=True)
233        elif task["type"] == "heading":
234            heading = task
235            heading["items"] = tasks(
236                type="to-do",
237                heading=heading["uuid"],
238                context_trashed=None,
239                include_items=True,
240                database=database,
241            )
242
243    if uuid:
244        result = result[0]
245
246    return result

249def areas(uuid=None, include_items=False, **kwargs):
250    """
251    Read areas into dicts.
252
253    Parameters
254    ----------
255    uuid : str or None, optional
256        Any valid uuid of an area. If None, then return all areas.
257
258    include_items : bool, default False
259        Include tasks and projects in each area.
260
261    tag : str or bool or None, optional
262        Any valid title of a tag. Only include areas matching that tag.
263        Special cases:
264        - `tag == False`, only include areas _without_ tags.
265        - `tag == True`, only include areas _with_ tags.
266        - `tag == None`, then ignore any tags present, that is,
267           include areas both with and without tags.
268
269    count_only : bool, default False
270        Only output length of result. This is done by a SQL COUNT query.
271
272    filepath : str, optional
273        Any valid path of a SQLite database file generated by the Things app.
274        If no path is provided, then access the default database path.
275
276    database : things.database.Database, optional
277        Any valid `things.database.Database` object previously instantiated.
278
279    Returns
280    -------
281    list of dict (default)
282        Representing Things areas.
283    dict (if `uuid` is given)
284        Representing a single Things area.
285    int (`count_only == True`)
286        Count of matching areas.
287
288    Examples
289    --------
290    >>> things.areas()
291    [{'uuid': 'Y3JC4XeyGWxzDocQL4aobo', 'type': 'area', 'title': 'Area 3'}, ...
292    >>> things.areas(tag='Home')
293    []
294    >>> things.areas(uuid='DciSFacytdrNG1nRaMJPgY')
295    {'uuid': 'DciSFacytdrNG1nRaMJPgY', 'type': 'area', 'title': 'Area 1', ...
296    >>> things.areas(include_items=True, tag='Errand')
297    [{'uuid': 'DciSFacytdrNG1nRaMJPgY', 'type': 'area', 'title': 'Area 1', ...
298    """
299    database = pop_database(kwargs)
300    result = database.get_areas(uuid=uuid, **kwargs)
301
302    if kwargs.get("count_only"):
303        return result
304
305    for area in result:
306        if area.get("tags"):
307            area["tags"] = database.get_tags(area=area["uuid"])
308        if include_items:
309            area["items"] = tasks(
310                area=area["uuid"], include_items=True, database=database
311            )
312
313    if uuid:
314        result = result[0]
315
316    return result

319def tags(title=None, include_items=False, **kwargs):
320    """
321    Read tags into dicts.
322
323    Parameters
324    ----------
325    title : str, optional
326        Any valid title of a tag. Include all items of said tag.
327        If None, then return all tags.
328
329    include_items : bool, default False
330        For each tag, include items tagged with that tag.
331        Items may include areas, tasks, and projects.
332
333    area : str, optional
334        Valid uuid of an area. Return tags of said area.
335
336    task : str, optional
337        Valid uuid of a task. Return tags of said task.
338
339    titles_only : bool, default False
340        If True, only return list of titles of tags.
341
342    filepath : str, optional
343        Any valid path of a SQLite database file generated by the Things app.
344        If no path is provided, then access the default database path.
345
346    database : things.database.Database, optional
347        Any valid database object previously instantiated.
348
349    Returns
350    -------
351    list of dict (default)
352        Representing tags.
353    list of str (if `titles_only == True` or area / task is given)
354        Representing tag titles.
355    dict (if `title` is given)
356        Representing a single Things tag.
357
358    Examples
359    --------
360    >>> things.tags()
361    [{'uuid': 'H96sVJwE7VJveAnv7itmux', 'type': 'tag', 'title': 'Errand', ...
362    >>> things.tags('Home')
363    {'uuid': 'CK9dARrf2ezbFvrVUUxkHE', 'type': 'tag', 'title': 'Home', ...
364    >>> things.tags(include_items=True)
365    [{'uuid': 'H96sVJwE7VJveAnv7itmux', 'type': 'tag', 'title': 'Errand', ...
366    >>> things.tags(task='2Ukg8I2nLukhyEM7wYiBeb')
367    []
368    """
369    database = pop_database(kwargs)
370    result = database.get_tags(title=title, **kwargs)
371
372    if include_items:
373        for tag in result:
374            tag_title = tag["title"]
375            tag["items"] = [
376                *areas(tag=tag_title, database=database),
377                *tasks(tag=tag_title, database=database),
378            ]
379
380    if title:
381        result = result[0]
382
383    return result

434def get(uuid, default=None, **kwargs):
435    """
436    Find an object by uuid. If not found, return `default`.
437
438    Currently supports tasks, projects, headings, areas, and tags.
439    """
440    try:
441        return tasks(uuid=uuid, **kwargs)
442    except ValueError:
443        pass
444
445    try:
446        return areas(uuid=uuid, **kwargs)
447    except ValueError:
448        pass
449
450    for tag in tags(**kwargs):
451        if tag["uuid"] == uuid:
452            return tag
453
454    return default

490def today(**kwargs):
491    """
492    Read Today's tasks into dicts.
493
494    Note: The Things database reflects the state of the Things app when
495    it was last opened. For the Today tasks that means the database
496    might not be up to date anymore if you didn't open the app recently.
497    To get around this limitation, we here make a prediction of what
498    tasks would show up in Today if you were to open the app right now.
499    This prediction does not include repeating tasks at this time.
500
501    See `things.api.tasks` for details on the optional parameters.
502    """
503    database = pop_database(kwargs)
504    kwargs["database"] = database
505
506    regular_today_tasks = tasks(
507        start_date=True,
508        start="Anytime",
509        index="todayIndex",
510        **kwargs,
511    )
512
513    # Predictions of new to-dos indicated by a yellow dot in the app.
514
515    unconfirmed_scheduled_tasks = tasks(
516        start_date="past",
517        start="Someday",
518        index="todayIndex",
519        **kwargs,
520    )
521
522    unconfirmed_overdue_tasks = tasks(
523        start_date=False,
524        deadline="past",
525        deadline_suppressed=False,
526        **kwargs,
527    )
528
529    result = [
530        *regular_today_tasks,
531        *unconfirmed_scheduled_tasks,
532        *unconfirmed_overdue_tasks,
533    ]
534    result.sort(key=lambda task: (task["today_index"], task["start_date"]))
535
536    return result

635def last(offset, **kwargs):
636    """
637    Read tasks created within last X days, weeks, or years into dicts.
638
639    Per default, only incomplete tasks are included, but do see
640    `things.api.tasks` for details on the optional parameters.
641
642    Parameters
643    ----------
644    offset : str
645        A valid date offset such as '3d', '5w', or '1y'.
646        For details, see the `last` parameter of `things.api.tasks`.
647
648    Returns
649    -------
650    list of dict
651        Tasks within offset ordered by creation date. Newest first.
652
653    Examples
654    --------
655    >>> things.last('3d')
656    []
657    >>> things.last('1w', status='completed')
658    []
659    >>> things.last('1y', tag='Important', status='completed', count_only=True)
660    0
661
662    """
663    if offset is None:
664        raise ValueError(f"Invalid offset type: {offset!r}")
665
666    result = tasks(last=offset, **kwargs)
667    if result:
668        result.sort(key=lambda task: task["created"], reverse=True)
669
670    return result
