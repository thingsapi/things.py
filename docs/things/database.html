186    def get_tasks(  # pylint: disable=R0914
187        self,
188        uuid=None,
189        type=None,  # pylint: disable=W0622
190        status=None,
191        start=None,
192        area=None,
193        project=None,
194        heading=None,
195        tag=None,
196        start_date=None,
197        stop_date=None,
198        exact=False,
199        deadline=None,
200        deadline_suppressed=None,
201        trashed=False,
202        context_trashed=False,
203        last=None,
204        search_query: str = "",
205        index="index",
206        count_only=False,
207    ):
208        """Get tasks. See `things.api.tasks` for details on parameters."""
209        if uuid:
210            return self.get_task_by_uuid(uuid, count_only=count_only)
211
212        # Overwrites
213        start = start and start.title()
214
215        # Validation
216        validate("deadline", deadline, [None] + list(DATES))
217        validate("deadline_suppressed", deadline_suppressed, [None, True, False])
218        validate("start", start, [None] + list(START_TO_FILTER))
219        validate("start_date", start_date, [None] + list(DATES))
220        validate("status", status, [None] + list(STATUS_TO_FILTER))
221        validate("trashed", trashed, [None] + list(TRASHED_TO_FILTER))
222        validate("type", type, [None] + list(TYPE_TO_FILTER))
223        validate("context_trashed", context_trashed, [None, True, False])
224        validate("index", index, list(INDICES))
225        validate_offset("last", last)
226
227        if tag is not None:
228            valid_tags = self.get_tags(titles_only=True)
229            validate("tag", tag, [None] + list(valid_tags))
230
231        # Query
232        # TK: might consider executing SQL with parameters instead.
233        # See: https://docs.python.org/3/library/sqlite3.html#sqlite3.Cursor.execute
234
235        start_filter: str = START_TO_FILTER.get(start, "") if start else ""
236        status_filter: str = STATUS_TO_FILTER.get(status, "") if status else ""
237        trashed_filter: str = get_allow_none(TRASHED_TO_FILTER, trashed, "")
238        type_filter: str = TYPE_TO_FILTER.get(type, "") if type else ""
239
240        # Sometimes a task is _not_ set to trashed, but its context
241        # (project or heading it is contained within) is set to trashed.
242        # In those cases, the task wouldn't show up in any app view
243        # except for "Trash".
244        project_trashed_filter = make_truthy_filter("PROJECT.trashed", context_trashed)
245        project_of_heading_trashed_filter = make_truthy_filter(
246            "PROJECT_OF_HEADING.trashed", context_trashed
247        )
248
249        where_predicate = f"""
250            TASK.{IS_NOT_RECURRING}
251            {trashed_filter and f"AND TASK.{trashed_filter}"}
252            {project_trashed_filter}
253            {project_of_heading_trashed_filter}
254            {type_filter and f"AND TASK.{type_filter}"}
255            {start_filter and f"AND TASK.{start_filter}"}
256            {status_filter and f"AND TASK.{status_filter}"}
257            {make_filter('TASK.uuid', uuid)}
258            {make_filter("TASK.area", area)}
259            {make_filter("TASK.project", project)}
260            {make_filter("TASK.heading", heading)}
261            {make_filter("TASK.deadlineSuppressionDate", deadline_suppressed)}
262            {make_filter("TAG.title", tag)}
263            {make_date_filter(f"TASK.{DATE_START}", start_date)}
264            {make_date_filter(f"TASK.{DATE_STOP}", stop_date, exact)}
265            {make_date_filter(f"TASK.{DATE_DEADLINE}", deadline)}
266            {make_date_range_filter(f"TASK.{DATE_CREATED}", last)}
267            {make_search_filter(search_query)}
268            """
269        order_predicate = f'TASK."{index}"'
270
271        sql_query = make_tasks_sql_query(where_predicate, order_predicate)
272
273        if count_only:
274            return self.get_count(sql_query)
275
276        return self.execute_query(sql_query)

278    def get_task_by_uuid(self, uuid, count_only=False):
279        """Get a task by uuid. Raise `ValueError` if not found."""
280        where_predicate = "TASK.uuid = ?"
281        sql_query = make_tasks_sql_query(where_predicate)
282        parameters = (uuid,)
283
284        if count_only:
285            return self.get_count(sql_query, parameters)
286
287        result = self.execute_query(sql_query, parameters)
288        if not result:
289            raise ValueError(f"No such task uuid found: {uuid!r}")
290
291        return result

293    def get_areas(self, uuid=None, tag=None, count_only=False):
294        """Get areas. See `api.areas` for details on parameters."""
295        # Validation
296        if tag is not None:
297            valid_tags = self.get_tags(titles_only=True)
298            validate("tag", tag, [None] + list(valid_tags))
299
300        if (
301            uuid
302            and count_only is False
303            and not self.get_areas(uuid=uuid, count_only=True)
304        ):
305            raise ValueError(f"No such area uuid found: {uuid!r}")
306
307        # Query
308        sql_query = f"""
309            SELECT DISTINCT
310                AREA.uuid,
311                'area' as type,
312                AREA.title,
313                CASE
314                    WHEN AREA_TAG.areas IS NOT NULL THEN 1
315                END AS tags
316            FROM
317                {TABLE_AREA} AS AREA
318            LEFT OUTER JOIN
319                {TABLE_AREATAG} AREA_TAG ON AREA_TAG.areas = AREA.uuid
320            LEFT OUTER JOIN
321                {TABLE_TAG} TAG ON TAG.uuid = AREA_TAG.tags
322            WHERE
323                TRUE
324                {make_filter('TAG.title', tag)}
325                {make_filter('AREA.uuid', uuid)}
326            ORDER BY AREA."index"
327            """
328
329        if count_only:
330            return self.get_count(sql_query)

332        return self.execute_query(sql_query)

455    def execute_query(self, sql_query, parameters=(), row_factory=None):
456        """Run the actual SQL query."""
457        if self.print_sql or self.debug:
458            if not hasattr(self, "execute_query_count"):
459                # This is needed for historical `self.debug`.
460                # TK: might consider removing `debug` flag.
461                self.execute_query_count = 0
462            self.execute_query_count += 1
463            if self.debug:
464                print(f"/* Filepath {self.filepath!r} */")
465            print(f"/* Query {self.execute_query_count} */")
466            if parameters:
467                print(f"/* Parameters: {parameters!r} */")
468            print()
469            print(prettify_sql(sql_query))
470            print()
471
472        # "ro" means read-only
473        # See: https://sqlite.org/uri.html#recognized_query_parameters
474        uri = f"file:{self.filepath}?mode=ro"
475        connection = sqlite3.connect(uri, uri=True)  # pylint: disable=E1101
476        connection.row_factory = row_factory or dict_factory
477        cursor = connection.cursor()
478        cursor.execute(sql_query, parameters)
479
480        return cursor.fetchall()

492def make_tasks_sql_query(where_predicate=None, order_predicate=None):
493    """Make SQL query for Task table."""
494    where_predicate = where_predicate or "TRUE"
495    order_predicate = order_predicate or 'TASK."index"'
496
497    # Note: see remark at `make_date_filter()` as for why the first
498    # two `date()` statements have no "localtime" modifier.
499    return f"""
500            SELECT DISTINCT
501                TASK.uuid,
502                CASE
503                    WHEN TASK.{IS_TODO} THEN 'to-do'
504                    WHEN TASK.{IS_PROJECT} THEN 'project'
505                    WHEN TASK.{IS_HEADING} THEN 'heading'
506                END AS type,
507                CASE
508                    WHEN TASK.{IS_TRASHED} THEN 1
509                END AS trashed,
510                TASK.title,
511                CASE
512                    WHEN TASK.{IS_INCOMPLETE} THEN 'incomplete'
513                    WHEN TASK.{IS_CANCELED} THEN 'canceled'
514                    WHEN TASK.{IS_COMPLETED} THEN 'completed'
515                END AS status,
516                CASE
517                    WHEN AREA.uuid IS NOT NULL THEN AREA.uuid
518                END AS area,
519                CASE
520                    WHEN AREA.uuid IS NOT NULL THEN AREA.title
521                END AS area_title,
522                CASE
523                    WHEN PROJECT.uuid IS NOT NULL THEN PROJECT.uuid
524                END AS project,
525                CASE
526                    WHEN PROJECT.uuid IS NOT NULL THEN PROJECT.title
527                END AS project_title,
528                CASE
529                    WHEN HEADING.uuid IS NOT NULL THEN HEADING.uuid
530                END AS heading,
531                CASE
532                    WHEN HEADING.uuid IS NOT NULL THEN HEADING.title
533                END AS heading_title,
534                TASK.notes,
535                CASE
536                    WHEN TAG.uuid IS NOT NULL THEN 1
537                END AS tags,
538                CASE
539                    WHEN TASK.{IS_INBOX} THEN 'Inbox'
540                    WHEN TASK.{IS_ANYTIME} THEN 'Anytime'
541                    WHEN TASK.{IS_SOMEDAY} THEN 'Someday'
542                END AS start,
543                CASE
544                    WHEN CHECKLIST_ITEM.uuid IS NOT NULL THEN 1
545                END AS checklist,
546                date(TASK.startDate, "unixepoch") AS start_date,
547                date(TASK.{DATE_DEADLINE}, "unixepoch") AS deadline,
548                date(TASK.stopDate, "unixepoch", "localtime") AS "stop_date",

578def dict_factory(cursor, row):
579    """
580    Convert SQL result into a dictionary.
581
582    See also:
583    https://docs.python.org/3/library/sqlite3.html#sqlite3.Connection.row_factory
584    """
585    result = {}
586    for index, column in enumerate(cursor.description):
587        key, value = column[0], row[index]
588        if value is None and key in COLUMNS_TO_OMIT_IF_NONE:
589            continue
590        if value and key in COLUMNS_TO_TRANSFORM_TO_BOOL:
591            value = bool(value)
592        result[key] = value
593    return result

596def escape_string(string):
597    r"""
598    Escape SQLite string literal.
599
600    Three notes:
601
602    1. A single quote within a SQLite string can be encoded by putting
603    two single quotes in a row. Escapes using the backslash character
604    are not supported in SQLite.
605
606    2. Null characters '\0' within strings can lead to surprising
607    behavior. However, `cursor.execute` will already throw a `ValueError`
608    if it finds a null character in the query, so we let it handle
609    this case for us.
610
611    3. Eventually we might want to make use of parameters instead of
612    manually escaping. Since this will require some refactoring,
613    we are going with the easiest solution for now.
614
615    See: https://www.sqlite.org/lang_expr.html#literal_values_constants_
616    """
617    return string.replace("'", "''")

625def make_filter(column, value):
626    """
627    Return SQL filter 'AND {column} = "{value}"'.
628
629    Special handling if `value` is `bool` or `None`.
630
631    Examples
632    --------
633    >>> make_filter('title', 'Important')
634    "AND title = 'Important'"
635
636    >>> make_filter('startDate', True)
637    'AND startDate IS NOT NULL'
638
639    >>> make_filter('startDate', False)
640    'AND startDate IS NULL'
641
642    >>> make_filter('title', None)
643    ''
644    """
645    default = f"AND {column} = '{escape_string(str(value))}'"
646    return {
647        None: "",
648        False: f"AND {column} IS NULL",
649        True: f"AND {column} IS NOT NULL",
650    }.get(value, default)

653def make_date_filter(date_column: str, value, exact=False) -> str:
654    """
655    Return a SQL filter for date columns.
656
657    Parameters
658    ----------
659    date_column : str
660        Name of the column that has date information on a task.
661
662    value : bool, 'future', 'past', ISO 8601 date str, or None
663        `True` or `False` indicates whether a date is set or not.
664        `'future'` or `'past'` indicates a date in the future or past.
665        ISO 8601 date str is in the format "YYYY-MM-DD".
666        `None` indicates any value.
667
668    exact : bool
669        matches for exact date of date_column
670
671    Returns
672    -------
673    str
674        A date filter for the SQL query. If `value == None`, then
675        return the empty string.
676
677    Examples
678    --------
679    >>> make_date_filter('startDate', True)
680    'AND startDate IS NOT NULL'
681
682    >>> make_date_filter('startDate', False)
683    'AND startDate IS NULL'
684
685    >>> make_date_filter('startDate', 'future')
686    "AND date(startDate, 'unixepoch') > date('now', 'localtime')"
687
688    >>> make_date_filter('stopDate', '2021-03-28')
689    "AND date(stopDate, 'unixepoch') >= date('2021-03-28')"
690
691    >>> make_date_filter('created', None)
692    ''
693
694    """
695    if value is None:
696        return ""
697
698    if isinstance(value, bool):
699        return make_filter(date_column, value)
700
701    try:
702        # Check for ISO 8601 date str
703        datetime.date.fromisoformat(value)
704        threshold = f"date('{value}')"
705        comparator = '==' if exact else '>='
706    except ValueError:
707        # "future" or "past"
708        validate("value", value, ["future", "past"])
709        threshold = "date('now', 'localtime')"
710        comparator = ">" if value == "future" else "<="
711
712    # Note: not using "localtime" modifier on `date()` since Things.app
713    # seems to store `startDate` and `dueDate` as a 00:00 UTC datetime.
714    # Morever, note that `stopDate` -- contrary to its name -- seems to
715    # be stored as the full UTC datetime of when the task was "stopped".
716    # See also: https://github.com/thingsapi/things.py/issues/93
717    date = f"date({date_column}, 'unixepoch')"
718
719    return f"AND {date} {comparator} {threshold}"

722def make_date_range_filter(date_column, offset) -> str:
723    """
724    Return a SQL filter to limit a date to last X days, weeks, or years.
725
726    Parameters
727    ----------
728    date_column : str
729        Name of the column that has date information on a task.
730
731    offset : str or None
732        A string comprised of an integer and a single character that can
733        be 'd', 'w', or 'y' that determines whether to return all tasks
734        for the past X days, weeks, or years.
735
736    Returns
737    -------
738    str
739        A date filter for the SQL query. If `offset == None`, then
740        return the empty string.
741
742    Examples
743    --------
744    >>> make_date_range_filter('created', '3d')
745    "AND datetime(created, 'unixepoch') > datetime('now', '-3 days')"
746
747    >>> make_date_range_filter('created', None)
748    ''
749    """
750    if offset is None:
751        return ""
752
753    validate_offset("offset", offset)
754    number, suffix = int(offset[:-1]), offset[-1]
755
756    if suffix == "d":
757        modifier = f"-{number} days"
758    elif suffix == "w":
759        modifier = f"-{number * 7} days"
760    elif suffix == "y":
761        modifier = f"-{number} years"
762
763    column_datetime = f"datetime({date_column}, 'unixepoch')"
764    offset_datetime = f"datetime('now', '{modifier}')"  # type: ignore
765
766    return f"AND {column_datetime} > {offset_datetime}"

769def make_truthy_filter(column: str, value) -> str:
770    """
771    Return a SQL filter that matches if a column is truthy or falsy.
772
773    Truthy means TRUE. Falsy means FALSE or NULL. This is akin
774    to how Python defines it natively.
775
776    Passing in `value == None` returns the empty string.
777
778    Examples
779    --------
780    >>> make_truthy_filter('PROJECT.trashed', True)
781    'AND PROJECT.trashed'
782
783    >>> make_truthy_filter('PROJECT.trashed', False)
784    'AND NOT IFNULL(PROJECT.trashed, 0)'
785
786    >>> make_truthy_filter('PROJECT.trashed', None)
787    ''
788    """
789    if value is None:
790        return ""
791
792    if value:
793        return f"AND {column}"
794
795    return f"AND NOT IFNULL({column}, 0)"
